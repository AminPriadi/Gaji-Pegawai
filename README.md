# Gaji-Pegawai
    #include<iostream>
    using namespace std;

    int main()
    {
        float gaji_perjam=5000;
        int gaji,jam;

        cout<<"masukan jumlah jam kerja seorang pegawai ";
        cin>>jam,

        gaji=jam*gaji_perjam;
        cout<<"maka jumlah gaji seorang pegawai adalah "<<gaji;

        return 0;
    }
